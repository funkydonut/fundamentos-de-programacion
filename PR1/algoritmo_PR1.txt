
const
    MAX_READINGS: integer = 100 {Maximum number of readings allowed}
    POWER_THRESHOLD: real = 0.0
end const

type
    tTariffType = {VALLEY=1, FLAT=2, PEAK=3} {Enum to represent tariff types}
end type

algorithm SmartGridAnalysis
    var
        powerData: vector[MAX_READINGS] of real; {Array for power values}
        tariffData: vector[MAX_READINGS] of integer; {Array for tariff types}
        discardedReadings: vector[MAX_READINGS] of real; {Array for discarded (invalid) power readings}
        totalConsumption: real; {Sum of valids readings}
        totalElements: integer; {Number of readings}
        validReadingsCount: integer; {Number of valid readings}
        invalidReadingsCount: integer; {Number of invalid readings}
        averagePower: real; {Average power of readings}
        peakPower: real; {Maximum power of readings}
        peakTariff: integer; {Maximum power tariff}
        valleyTariffCounter: integer; 
        flatTariffCounter: integer;
        peakTariffCounter: integer;
        i, j: integer;
        filename: string; {Name of the file to read}
        fileToRead: file; {File to open}
    end var
    
    {Data input from file}
    writeString("--- Smart Grid Analysis System (PR1) ---");
    writeString("LOAD DATA FROM FILE. ENTER FILE NAME >> ");
    filename := readString();
    
    fileToRead := openFile(filename);
    
    {Read number of readings from file}
    totalElements := readIntegerFromFile(fileToRead: file): integer; {apuntes PEC3 entrada y salida de datos}
    
    {Ensure we don't exceed array capacity}
    if totalElements > MAX_READINGS then
        totalElements := MAX_READINGS;
    end if
    
    {Read power data and tariffs from file}
    for i := 0 to totalElements - 1 do
        powerData[i] := readRealFromFile(fileToRead);
        tariffData[i] := readIntegerFromFile(fileToRead);
    end for
    
    closeFile(fileToRead);
    
    {Data Processing}
    writeString("Processing ");
    writeInteger(totalElements);
    writeString(" records...");
    
    {Initialize counters}
    totalConsumption := 0.0;
    averagePower := 0.0;
    peakPower := 0.0; 
    validReadingsCount := 0;
    invalidReadingsCount := 0;
    valleyTariffCounter := 0; 
    flatTariffCounter := 0;
    peakTariffCounter := 0; 
    
    for j := 0 to totalElements - 1 do
        {Process valid/non-valid readings}
        if powerData[j] is < POWER_THRESHOLD then
            {Register non-valid readings}
            discardedReadings[invalidReadingsCount] := powerData[j];
            invalidReadingsCount := invalidReadingsCount + 1;
        else
            {Register valid readings and calculate total consumption}
            validReadingsCount := validReadingsCount + 1; 
            totalConsumption := totalConsumption + powerData[j]; 
            {Check for maximum peak power among valid readings}
            if (validReadingsCount = 1) or (powerData[j] > peakPower) then
                peakPower := powerData[j];
                peakTariff := tariffData[j];
            end if
            {Update tariff counters}
            switch tariffData[j] {Switch mejor if...else}
                case tariffData[j] = VALLEY then
                    valleyTariffCounter := valleyTariffCounter + 1;
                end case
                case tariffData[j] = FLAT then
                    flatTariffCounter := flatTariffCounter + 1;
                end case
                case tariffData[j] = PEAK then
                    peakTariffCounter := peakTariffCounter + 1;
                end case
            end switch         
        end if
    end for
    
    {Data Output}
    writeString("----------------------------------------");
    writeString("ANALYSIS SUMMARY");
    writeString("----------------------------------------");
    
    if validReadingsCount > 0 then
        {Calculate average power}
        averagePower := totalConsumption / validReadingsCount; 
        {Output stats and distribution}
        writeString("Total consumption: ");
        writeReal(totalConsumption);
        writeString(" kW");
        writeString("Average power: ");
        writeReal(averagePower);
        writeString(" kW");
        writeString("Maximum peak: ");
        writeReal(peakPower);
        writeString(" kW (Tariff: ");
        if peakTariff = VALLEY then
            writeString("VALLEY");
        else
            if peakTariff = FLAT then
                writeString("FLAT");
            else
                writeString("PEAK");
            end if
        end if
        writeString(")");
        writeString("Distribution by Tariff: ");
        writeString("- Valley intervals: ");
        writeInteger(valleyTariffCounter); 
        writeString("- Flat intervals: ");
        writeInteger(flatTariffCounter); 
        writeString("- Peak intervals: ");
        writeInteger(peakTariffCounter);
        writeString("Discarded readings: ");
        writeInteger(invalidReadingsCount);
        {Output discarded readings list (one per line)} {No estaba en el enunciado, pero si en los ejemplos de impresión. Explicar de forma somera.}
        if invalidReadingsCount > 0 then
            for i := 0 to invalidReadingsCount - 1 do
                writeReal(discardedReadings[i]); 
                writeString(" kW");
            end for
        end if
        
        writeString("----------------------------------------");
    else
        writeString("No valid data found to perform analysis.");
    end if
end algorithm
